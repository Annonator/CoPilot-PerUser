# Copilot AI Usage Scaffold Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Scaffold a Docker-capable Next.js plus Go app where Google-authenticated users can view only their own GitHub Enterprise AI credit usage through a server-side Go proxy.

**Architecture:** The frontend in `src/web` owns Google Auth.js login and mints a short-lived internal app JWT for the Go API. The backend in `src/api` validates that token, resolves company email to GitHub login through an identity resolver interface, calls GitHub Enterprise AI credit billing APIs through a GitHub client, normalizes the response, and returns self-only dashboard data. Docker Compose runs the two services locally without a database.

**Tech Stack:** Next.js App Router, Auth.js / `next-auth@beta`, React, TypeScript, Go standard library HTTP server, standard-library HMAC JWT verification, Docker, Docker Compose.

---

## Scope Check

This plan implements the first scaffold only. It creates a working local foundation with fixture/static identity support, API contracts, tests, Docker runtime, and a dashboard shell. It does not implement a production SAML identity crawler because the exact GitHub enterprise identity API call must be proven with real admin credentials; the resolver boundary is created so that live implementation can be added behind the same interface.

```go
package config

import (
	"fmt"
	"os"
	"strings"
	"time"
)

type Config struct {
	Port                        string
	CompanyEmailDomains         []string
	AppTokenSecret              string
	GitHubAPIBaseURL            string
	GitHubEnterpriseSlug        string
	GitHubAdminToken            string
	GitHubIdentityResolver      string
	GitHubIdentityStaticMapPath string
	UsageCacheTTL              time.Duration
}

func Load() (Config, error) {
	cfg := Config{
		Port:                        envDefault("PORT", "8080"),
		CompanyEmailDomains:         splitCSV(os.Getenv("COMPANY_EMAIL_DOMAINS")),
		AppTokenSecret:              os.Getenv("APP_TOKEN_SECRET"),
		GitHubAPIBaseURL:            envDefault("GITHUB_API_BASE_URL", "https://api.github.com"),
		GitHubEnterpriseSlug:        os.Getenv("GITHUB_ENTERPRISE_SLUG"),
		GitHubAdminToken:            os.Getenv("GITHUB_ADMIN_TOKEN"),
		GitHubIdentityResolver:      envDefault("GITHUB_IDENTITY_RESOLVER", "static"),
		GitHubIdentityStaticMapPath: os.Getenv("GITHUB_IDENTITY_STATIC_MAP_PATH"),
		UsageCacheTTL:              10 * time.Minute,
	}

	if rawTTL := os.Getenv("USAGE_CACHE_TTL"); rawTTL != "" {
		ttl, err := time.ParseDuration(rawTTL)
		if err != nil {
			return Config{}, fmt.Errorf("parse USAGE_CACHE_TTL: %w", err)
		}
		cfg.UsageCacheTTL = ttl
	}

	if len(cfg.CompanyEmailDomains) == 0 {
		return Config{}, fmt.Errorf("COMPANY_EMAIL_DOMAINS is required")
	}
	if cfg.AppTokenSecret == "" {
		return Config{}, fmt.Errorf("APP_TOKEN_SECRET is required")
	}
	if cfg.GitHubEnterpriseSlug == "" {
		return Config{}, fmt.Errorf("GITHUB_ENTERPRISE_SLUG is required")
	}
	if cfg.GitHubAdminToken == "" {
		return Config{}, fmt.Errorf("GITHUB_ADMIN_TOKEN is required")
	}
	if cfg.GitHubIdentityResolver == "static" && cfg.GitHubIdentityStaticMapPath == "" {
		return Config{}, fmt.Errorf("GITHUB_IDENTITY_STATIC_MAP_PATH is required for static resolver")
	}

	return cfg, nil
}

func envDefault(key string, fallback string) string {
	if value := os.Getenv(key); value != "" {
		return value
	}
	return fallback
}

func splitCSV(raw string) []string {
	var out []string
	for _, part := range strings.Split(raw, ",") {
		part = strings.TrimSpace(strings.ToLower(part))
		if part != "" {
			out = append(out, part)
		}
	}
	return out
}
```

```go
package auth

import (
	"crypto/hmac"
	"crypto/sha256"
	"encoding/base64"
	"encoding/json"
	"errors"
	"fmt"
	"strings"
	"time"
)

type Claims struct {
	Email string `json:"email"`
	Name  string `json:"name,omitempty"`
	Exp   int64  `json:"exp"`
}

type Manager struct {
	Secret []byte
	Now    func() time.Time
}

func (m Manager) Sign(claims Claims, ttl time.Duration) (string, error) {
	now := m.now()
	claims.Exp = now.Add(ttl).Unix()

	header, err := encodeJSON(map[string]string{"alg": "HS256", "typ": "JWT"})
	if err != nil {
		return "", err
	}
	payload, err := encodeJSON(claims)
	if err != nil {
		return "", err
	}
	signingInput := header + "." + payload
	return signingInput + "." + m.signature(signingInput), nil
}

func (m Manager) Validate(authorization string) (Claims, error) {
	if !strings.HasPrefix(authorization, "Bearer ") {
		return Claims{}, errors.New("missing bearer token")
	}
	token := strings.TrimPrefix(authorization, "Bearer ")
	parts := strings.Split(token, ".")
	if len(parts) != 3 {
		return Claims{}, errors.New("invalid token format")
	}
	signingInput := parts[0] + "." + parts[1]
	if !hmac.Equal([]byte(parts[2]), []byte(m.signature(signingInput))) {
		return Claims{}, errors.New("invalid token signature")
	}
	var claims Claims
	payload, err := base64.RawURLEncoding.DecodeString(parts[1])
	if err != nil {
		return Claims{}, fmt.Errorf("decode payload: %w", err)
	}
	if err := json.Unmarshal(payload, &claims); err != nil {
		return Claims{}, fmt.Errorf("unmarshal payload: %w", err)
	}
	if claims.Email == "" {
		return Claims{}, errors.New("email claim is required")
	}
	if claims.Exp <= m.now().Unix() {
		return Claims{}, errors.New("token is expired")
	}
	return claims, nil
}

func AllowedDomain(email string, domains []string) bool {
	at := strings.LastIndex(email, "@")
	if at < 0 {
		return false
	}
	domain := strings.ToLower(email[at+1:])
	for _, allowed := range domains {
		if domain == strings.ToLower(strings.TrimSpace(allowed)) {
			return true
		}
	}
	return false
}

func (m Manager) now() time.Time {
	if m.Now != nil {
		return m.Now()
	}
	return time.Now()
}

func (m Manager) signature(signingInput string) string {
	mac := hmac.New(sha256.New, m.Secret)
	_, _ = mac.Write([]byte(signingInput))
	return base64.RawURLEncoding.EncodeToString(mac.Sum(nil))
}

func encodeJSON(value any) (string, error) {
	data, err := json.Marshal(value)
	if err != nil {
		return "", err
	}
	return base64.RawURLEncoding.EncodeToString(data), nil
}
```

```go
package identity

import (
	"context"
	"encoding/json"
	"fmt"
	"os"
	"strings"
)

type Resolver interface {
	ResolveGitHubLogin(ctx context.Context, email string) (string, error)
}

type StaticResolver struct {
	byEmail map[string]string
}

func NewStaticResolver(path string) (*StaticResolver, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("read static identity map: %w", err)
	}
	var raw map[string]string
	if err := json.Unmarshal(data, &raw); err != nil {
		return nil, fmt.Errorf("parse static identity map: %w", err)
	}
	normalized := make(map[string]string, len(raw))
	for email, login := range raw {
		normalized[strings.ToLower(strings.TrimSpace(email))] = strings.TrimSpace(login)
	}
	return &StaticResolver{byEmail: normalized}, nil
}

func (r *StaticResolver) ResolveGitHubLogin(_ context.Context, email string) (string, error) {
	login, ok := r.byEmail[strings.ToLower(strings.TrimSpace(email))]
	if !ok || login == "" {
		return "", fmt.Errorf("no GitHub login for email %q", email)
	}
	return login, nil
}
```

```go
import (
	"context"
	"encoding/json"
	"fmt"
	"net/http"
	"net/url"
	"strconv"
	"strings"
)

type BillingClient struct {
	baseURL string

}

type AICreditUsageRequest struct {
	Enterprise   string
	User         string
	Organization string
	Model        string

}

type AICreditUsageReport struct {
	TimePeriod struct {
		Year  int `json:"year"`
		Month int `json:"month,omitempty"`
		Day   int `json:"day,omitempty"`
	} `json:"timePeriod"`
	Enterprise string              `json:"enterprise"`

func NewBillingClient(baseURL string, token string, client *http.Client) *BillingClient {
	if client == nil {
		client = http.DefaultClient
	}
	return &BillingClient{
		baseURL: strings.TrimRight(baseURL, "/"),
		token:   token,
		client:  client,
	}
}

func (c *BillingClient) GetAICreditUsage(ctx context.Context, req AICreditUsageRequest) (AICreditUsageReport, error) {
	endpoint, err := url.Parse(c.baseURL + "/enterprises/" + url.PathEscape(req.Enterprise) + "/settings/billing/ai_credit/usage")
	if err != nil {
		return AICreditUsageReport{}, err
	}
	query := endpoint.Query()
	if req.Month > 0 {
		query.Set("month", strconv.Itoa(req.Month))
	}
	if req.User != "" {
		query.Set("user", req.User)
	}
	if req.Year > 0 {
		query.Set("year", strconv.Itoa(req.Year))
	}
	if req.Day > 0 {
		query.Set("day", strconv.Itoa(req.Day))
	}
	if req.Organization != "" {
		query.Set("organization", req.Organization)
	}
	if req.Model != "" {
		query.Set("model", req.Model)
	}
	if req.Product != "" {
		query.Set("product", req.Product)
	}
	endpoint.RawQuery = query.Encode()

	httpReq, err := http.NewRequestWithContext(ctx, http.MethodGet, endpoint.String(), nil)
	if err != nil {
		return AICreditUsageReport{}, err
	}
	httpReq.Header.Set("Accept", "application/vnd.github+json")
	httpReq.Header.Set("Authorization", "Bearer "+c.token)
	httpReq.Header.Set("X-GitHub-Api-Version", "2026-03-10")

	resp, err := c.client.Do(httpReq)
	if err != nil {
		return AICreditUsageReport{}, err
	}
	defer resp.Body.Close()

	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		return AICreditUsageReport{}, fmt.Errorf("github billing API returned %d", resp.StatusCode)
	}

	var report AICreditUsageReport
	if err := json.NewDecoder(resp.Body).Decode(&report); err != nil {
		return AICreditUsageReport{}, fmt.Errorf("decode billing usage: %w", err)
	}
	return report, nil
}
```

```go
package usage

import (
	"context"
	"fmt"
	"sync"
	"time"

	"copilot-per-user/api/internal/identity"
	gh "copilot-per-user/api/internal/github"
)

type BillingClient interface {
	GetAICreditUsage(ctx context.Context, req gh.AICreditUsageRequest) (gh.AICreditUsageReport, error)
}

type ServiceConfig struct {
	Enterprise string
	Resolver   identity.Resolver
	Billing    BillingClient
	CacheTTL   time.Duration
	Now        func() time.Time
}

type Service struct {
	cfg   ServiceConfig
	mu    sync.Mutex
	cache map[string]cacheEntry
}

type cacheEntry struct {
	expires time.Time
	value   MonthlyUsage
}

func NewService(cfg ServiceConfig) *Service {
	if cfg.CacheTTL == 0 {
		cfg.CacheTTL = 10 * time.Minute
	}
	if cfg.Now == nil {
		cfg.Now = time.Now
	}
	return &Service{cfg: cfg, cache: map[string]cacheEntry{}}
}

func (s *Service) GetMonthlyUsage(ctx context.Context, email string, year int, month int) (MonthlyUsage, error) {
	login, err := s.cfg.Resolver.ResolveGitHubLogin(ctx, email)
	if err != nil {
		return MonthlyUsage{}, err
	}
	key := fmt.Sprintf("%s:%s:%04d-%02d", s.cfg.Enterprise, login, year, month)
	if cached, ok := s.getCached(key); ok {
		cached.SourceMetadata.Cached = true
		return cached, nil
	}

	monthly, err := s.cfg.Billing.GetAICreditUsage(ctx, gh.AICreditUsageRequest{
		Enterprise: s.cfg.Enterprise,
		User:       login,
		Year:       year,
		Month:      month,
	})
	if err != nil {
		return MonthlyUsage{}, err
	}
	result := MonthlyUsage{
		Period: Period{Year: year, Month: month},
		User:   User{Email: email, GitHubLogin: login},
		Models: normalizeModels(monthly.UsageItems),
		SourceMetadata: SourceMetadata{
			Enterprise: s.cfg.Enterprise,
			Source:     "github_enterprise_billing_ai_credit_usage",
		},
	}
	result.Totals = sumModels(result.Models)

	for day := 1; day <= s.daysToFetch(year, month); day++ {
		daily, err := s.cfg.Billing.GetAICreditUsage(ctx, gh.AICreditUsageRequest{
			Enterprise: s.cfg.Enterprise,
			User:       login,
			Year:       year,
			Month:      month,
			Day:        day,
		})
		if err != nil {
			return MonthlyUsage{}, err
		}
		models := normalizeModels(daily.UsageItems)
		result.Daily = append(result.Daily, DailyUsage{
			Day:    fmt.Sprintf("%04d-%02d-%02d", year, month, day),
			Models: models,
			Totals: sumModels(models),
		})
	}

	s.setCached(key, result)
	return result, nil
}

func (s *Service) daysToFetch(year int, month int) int {
	now := s.cfg.Now()
	if now.Year() == year && int(now.Month()) == month {
		return now.Day()
	}
	return time.Date(year, time.Month(month)+1, 0, 0, 0, 0, 0, time.UTC).Day()
}

func (s *Service) getCached(key string) (MonthlyUsage, bool) {
	s.mu.Lock()
	defer s.mu.Unlock()
	entry, ok := s.cache[key]
	if !ok || s.cfg.Now().After(entry.expires) {
		return MonthlyUsage{}, false
	}
	return entry.value, true
}

func (s *Service) setCached(key string, value MonthlyUsage) {
	s.mu.Lock()
	defer s.mu.Unlock()
	s.cache[key] = cacheEntry{expires: s.cfg.Now().Add(s.cfg.CacheTTL), value: value}
}

func normalizeModels(items []gh.AICreditUsageItem) []ModelUsage {
	models := make([]ModelUsage, 0, len(items))
	for _, item := range items {
		models = append(models, ModelUsage{
			Model:             item.Model,
			IncludedCredits:   item.DiscountQuantity,
			AdditionalCredits: item.NetQuantity,
			GrossAmount:       item.GrossAmount,
			AdditionalUsage:   item.NetAmount,
			PricePerCredit:    item.PricePerUnit,
		})
	}
	return models
}

func sumModels(models []ModelUsage) UsageTotals {
	var totals UsageTotals
	for _, model := range models {
		totals.IncludedCredits += model.IncludedCredits
		totals.AdditionalCredits += model.AdditionalCredits
		totals.GrossAmount += model.GrossAmount
		totals.AdditionalUsage += model.AdditionalUsage
	}
	return totals
}
```

```go
package httpapi

import (
	"context"
	"encoding/json"
	"net/http"
	"strconv"

	"copilot-per-user/api/internal/auth"
	"copilot-per-user/api/internal/usage"
)

type UsageService interface {
	GetMonthlyUsage(ctx context.Context, email string, year int, month int) (usage.MonthlyUsage, error)
}

type ServerConfig struct {
	Auth                auth.Manager
	CompanyEmailDomains []string
	Usage               UsageService
}

type Server struct {
	mux *http.ServeMux
	cfg ServerConfig
}

func NewServer(cfg ServerConfig) *Server {
	s := &Server{mux: http.NewServeMux(), cfg: cfg}
	s.mux.HandleFunc("GET /healthz", s.healthz)
	s.mux.HandleFunc("GET /v1/me", s.me)
	s.mux.HandleFunc("GET /v1/usage", s.usage)
	return s
}

func (s *Server) ServeHTTP(w http.ResponseWriter, r *http.Request) {
	s.mux.ServeHTTP(w, r)
}

func (s *Server) healthz(w http.ResponseWriter, _ *http.Request) {
	writeJSON(w, http.StatusOK, map[string]string{"status": "ok"})
}

func (s *Server) me(w http.ResponseWriter, r *http.Request) {
	claims, ok := s.authenticate(w, r)
	if !ok {
		return
	}
	writeJSON(w, http.StatusOK, map[string]string{"email": claims.Email, "name": claims.Name})
}

func (s *Server) usage(w http.ResponseWriter, r *http.Request) {
	claims, ok := s.authenticate(w, r)
	if !ok {
		return
	}
	year, err := strconv.Atoi(r.URL.Query().Get("year"))
	if err != nil || year < 2000 {
		writeError(w, http.StatusBadRequest, "valid year is required")
		return
	}
	month, err := strconv.Atoi(r.URL.Query().Get("month"))
	if err != nil || month < 1 || month > 12 {
		writeError(w, http.StatusBadRequest, "valid month is required")
		return
	}
	result, err := s.cfg.Usage.GetMonthlyUsage(r.Context(), claims.Email, year, month)
	if err != nil {
		writeError(w, http.StatusBadGateway, "usage lookup failed")
		return
	}
	writeJSON(w, http.StatusOK, result)
}

func (s *Server) authenticate(w http.ResponseWriter, r *http.Request) (auth.Claims, bool) {
	claims, err := s.cfg.Auth.Validate(r.Header.Get("Authorization"))
	if err != nil {
		writeError(w, http.StatusUnauthorized, "authentication required")
		return auth.Claims{}, false
	}
	if !auth.AllowedDomain(claims.Email, s.cfg.CompanyEmailDomains) {
		writeError(w, http.StatusForbidden, "email domain is not allowed")
		return auth.Claims{}, false
	}
	return claims, true
}

func writeError(w http.ResponseWriter, status int, message string) {
	writeJSON(w, status, map[string]string{"error": message})
}

func writeJSON(w http.ResponseWriter, status int, value any) {
	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(status)
	_ = json.NewEncoder(w).Encode(value)
}
```

```ts
import { SignJWT } from "jose"

export type AppTokenUser = {
  email: string
  name?: string | null
}

export async function createAppToken(user: AppTokenUser): Promise<string> {
  const secret = process.env.APP_TOKEN_SECRET
  if (!secret) {
    throw new Error("APP_TOKEN_SECRET is required")
  }
  return new SignJWT({ email: user.email, name: user.name ?? "" })
    .setProtectedHeader({ alg: "HS256", typ: "JWT" })
    .setExpirationTime("5m")
    .sign(new TextEncoder().encode(secret))
}
```

```ts
function allowedDomains(): string[] {
```

```ts
import { auth } from "@/auth"
import { createAppToken } from "@/lib/app-token"
import type { MonthlyUsage } from "@/lib/usage-types"

export async function getCurrentUsage(year: number, month: number): Promise<MonthlyUsage> {
  const session = await auth()
  const email = session?.user?.email
  if (!email) {
    throw new Error("No authenticated email")
  }
  const token = await createAppToken({ email, name: session.user?.name })
  const baseUrl = process.env.API_BASE_URL ?? "http://localhost:8080"
  const response = await fetch(`${baseUrl}/v1/usage?year=${year}&month=${month}`, {
    headers: { Authorization: `Bearer ${token}` },
    cache: "no-store"
  })
  if (!response.ok) {
    throw new Error(`API returned ${response.status}`)
  }
  return response.json()
}
```
